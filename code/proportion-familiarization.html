Data

This data table contains the proportion of pigs that opened the door across 5 training days and the cumulative score.

Adjust data types

dat <- dat %>%
  mutate_at(vars(familiarization_day,
                 level), ~as.factor(.)) %>% 
  mutate(level = relevel(level, "group"))

Plots

plot <- dat %>%
  ggplot(aes(x    = familiarization_day,
             col   = level,
             shape = level)) +
  geom_point(aes(y = prop),
             size = 3,
             position = position_dodge(width = 0.1)) +
  geom_line(aes(y        = prop,
                group    = level,
                linetype = level),
            position = position_dodge(width = 0.1)) +
  geom_errorbar(aes(ymin = prop - group.sd,
                    ymax = prop + group.sd),
                width = 0.1,
                position = position_dodge(width = 0.1)) +
  scale_color_brewer(palette = "Set1")+
  scale_y_continuous(lim = c(0, 1.05), breaks = seq(0, 1, 0.25)) +
  labs(x = "Familiarization day",
       y = "Proportion of pigs that open a door") +
  my_theme +
  theme(legend.position = "right",
        legend.title = element_blank(),
        legend.text = element_text(size=12))
