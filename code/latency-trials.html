Plot

plot <- dat %>% 
  ggplot() + 
  geom_boxplot(data = subset(dat, helper == "Pig"),
               aes(x = as.factor(trapped.order), 
                   y = lat.helped.min),
               outlier.shape = NA, width = 0.5) + 
  geom_jitter(aes(x     = as.factor(trapped.order), 
                  y     = lat.helped.min,
                  col   = id.group,
                  shape = helper), 
              size = 3, width = 0.15, alpha = 0.8) +
  scale_shape_manual(values = c(17, 19),
                     labels = c("=  pig released by human", "")) +
  scale_color_manual(values = c(viridis(n=8)),
                     guide="none") +
  scale_y_continuous(lim = c(0, 20.01), breaks = seq(0, 20, 5)) +
  labs(x = "Group testing order",
       y = "Latency for helper to release trapped pig (min)",
       shape = "") +
  my_theme +
  guides(shape = guide_legend(override.aes = list(size  = 4,
                                                  shape = c(17, NA)))) +
  theme(legend.justification = c(1,0.6),
        legend.position = c(1,0.6),
        legend.background = element_rect(fill = NA, colour ="black",
                                         size=0.3, linetype="solid"),
        legend.text = element_text(size=12))
